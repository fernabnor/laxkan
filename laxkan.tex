  \section{Introduction}
  \section{Right Kan extensions of \texorpdfstring{$\bcat{C}\!\on{at}_\infty$-}-valued functors}
  For this section we fix a functor of marked $\infty$-bicategories $f: \CC^\dagger \to \DD^\dagger$.

  \begin{definition}
  	Given a marked $\infty$-bicategory $\CC^\dagger$ we define a $\msSet$-enriched category $\coCart(\CC)^\dagger$ as follows:
  	\begin{itemize}
  			\item Objects are given by maps of marked-scaled simplicial sets $p: X \to \CC$ such that $p$ is an inner coCartesian 1-fibration.
  			\item Given a pair of inner coCartesian fibrations $p: X \to \CC$ and $\pi: Y \to \CC$ we define a marked simplicial set of maps as the full subcategory $\on{Map}^\dagger_{\CC}(X,Y) \subset \on{Map}^\flat_{\CC}(X,Y)$ (see REF) consisting in those commutative diagrams
  			\[
  				\begin{tikzcd}
  					X \arrow[dr,"p",swap] \arrow[rr,"f"] && Y \arrow[dl,"\pi"] \\
  					& \CC &
  				\end{tikzcd}
  			\]
  			that preserve coCartesian edges lying over marked edge in $\CC^\dagger$.
  	\end{itemize}
  \end{definition}

  \begin{remark}
  	Observe that $\coCart(\CC)^\dagger$ is a fibrant $\msSet$-enriched category and thus defines an $(\infty,2)$-category. 
  \end{remark}

  \begin{remark}
  	Observe that given a marked functor $f:\CC^\dagger \to \DD^\dagger$ of $\infty$-bicategories we obtain a restriction functor
  	\[
  		\begin{tikzcd}
  			f^*: \coCart(\DD)^\dagger \arrow[r] & \coCart(\CC)^\dagger
  		\end{tikzcd}
  	\]
  	which is given by pullback along $f$.
  \end{remark}

 \begin{definition}
 	Given an inner coCartesian 1-fibration $p: X \to \CC$ we define a marked-scaled simplicial set  over $\DD$, $\on{R}(p):\mathcal{R}_X \to \DD$ as follows:
 	\begin{itemize}
 		\item  A simplex $\sigma: \Delta^n \to \mathcal{R}_X$ is given by a simplex $\alpha: \Delta^n \to \DD$ together with a map over $\CC$
 		\[\begin{tikzcd}
	{\Delta^n \times \alpha^*\left(\CC^\dagger_{\DD\upslash}\right)} && X \\
	& \CC
	\arrow[from=1-1, to=1-3,]
	\arrow[curve={height=12pt}, from=1-1, to=2-2,"\varphi",swap]
	\arrow[from=1-3, to=2-2,"p"]
\end{tikzcd}\]
where $\alpha^*\left(\CC^\dagger_{\DD\upslash}\right)$ is given by the pullback of marked-scaled simplicial sets
\[
	\begin{tikzcd}
		\alpha^*\left(\CC^\dagger_{\DD\upslash}\right) \arrow[r] \arrow[d] & \CC^\dagger_{\DD \upslash} \arrow[d] \\
		\Delta^n \arrow[r,"\alpha"] & \DD
	\end{tikzcd}
\]
and $\varphi$ is the composite  $\begin{tikzcd}
	\Delta^n \times \alpha^*\left(\CC^\dagger_{\DD\upslash}\right) \arrow[r] & \alpha^*\left(\CC^\dagger_{\DD\upslash}\right) \arrow[r] & \CC^{\dagger}_{\DD \upslash} \arrow[r] & \CC
\end{tikzcd}$
\item An edge $e: \Delta^1 \to \mathcal{R}_X$ is marked if and only if the corresponding map $\Delta^1 \times e^*\left(\CC^\dagger_{\DD\upslash}\right) \to X$ factors through $(\Delta^1)^\sharp \times e^*\left(\CC^\dagger_{\DD\upslash}\right)$.
\item We declare a triangle $\theta: \Delta^2 \to \mathcal{R}_X$ to be thin if and only if its associated simplex $\beta:\Delta^2 \to \DD$ is thin in $\DD$.
 	\end{itemize}

\begin{proposition}
	For every inner coCartesian 1-fibration $p: X \to \CC$ the map $\on{R}(p): \mathcal{R}_X \to \DD$ defines an inner coCartesian 1-fibration over $\DD$. Moreover, this assignment extends to a $\msSet$-enriched functor
	\[
		\begin{tikzcd}
			\mathfrak{R}f:\coCart(\DD)^\dagger \arrow[r] & \coCart(\CC)^\dagger.
		\end{tikzcd}
	\]
\end{proposition}

 \end{definition}
